33#include <shark/Statistics/Statistics.h>
34#include <shark/Core/Random.h>
35
36using namespace shark;
37
38
39int main(int argc, char** argv)
40{
41 statistics::ResultTable<double> table(2,"VarianceOfGaussian");//set a name for the results
42 table.setDimensionName(0,"input1");
43 table.setDimensionName(1,"input2");
44
45 // Fill the table with randomly generated numbers for different variances and mean and also add missing values
46 for(std::size_t k = 1; k != 10; ++k){
47 double var= 10.0*k;
48 for (std::size_t i = 0; i < 10000; i++){
49 double value1=random::gauss(random::globalRng, 0,var);
50 double value2=random::gauss(random::globalRng, 0,var);
51 if(random::coinToss(random::globalRng) == 1)
52 value2=statistics::missingValue();
53 table.update(var,value1,value2 );
54 }
55 }
56
57 statistics::Statistics<double> stats(&table);
58 stats.addStatistic(statistics::Mean());//adds a statistic "Mean" for each variable
59 stats.addStatistic("Variance", statistics::Variance());//explicit name
60 stats.addStatistic("Missing", statistics::FractionMissing());
61
62 printCSV(stats);
63}
