48namespace shark {namespace benchmarks{
49/*! \brief Multi-objective optimization benchmark function CIGTAB 1.
50*
51* The function is described in
52*
53* Christian Igel, Nikolaus Hansen, and Stefan Roth.
54* Covariance Matrix Adaptation for Multi-objective Optimization.
55* Evolutionary Computation 15(1), pp. 1-28, 2007
56* \ingroup benchmarks
57*/
58struct CIGTAB1 : public MultiObjectiveFunction {
59
60 CIGTAB1(std::size_t numberOfVariables = 5) : m_a( 1E6 ) {
61 m_features |= CAN_PROPOSE_STARTING_POINT;
62 m_numberOfVariables = numberOfVariables;
63 }
64
65 /// \brief From INameable: return the class name.
66 std::string name() const
67 { return "CIGTAB1"; }
68
69 std::size_t numberOfObjectives()const{
70 return 2;
71 }
72
73 std::size_t numberOfVariables()const{
74 return m_numberOfVariables;
75 }
76
77 bool hasScalableDimensionality()const{
78 return true;
79 }
80
81 /// \brief Adjusts the number of variables if the function is scalable.
82 /// \param [in] numberOfVariables The new dimension.
83 void setNumberOfVariables( std::size_t numberOfVariables ){
84 m_numberOfVariables = numberOfVariables;
85 }
86
87 void init() {
88 m_rotationMatrix = blas::randomRotationMatrix(*mep_rng, m_numberOfVariables);
89 }
90
91 ResultType eval( const SearchPointType & x ) const {
92 m_evaluationCounter++;
93
94 ResultType value(2);
95
96 ResultType y = prod( m_rotationMatrix, x );
97 double result = sqr( y(0) ) + sqr( m_a ) * sqr( y( numberOfVariables() - 1 ) );
98
99 for (unsigned i = 1; i < numberOfVariables() - 1; i++) {
100 result += m_a * sqr( y( i ) );
101 }
102
103 value[0] = result / ( sqr(m_a) * numberOfVariables() );
104
105 result = sqr(y( 0 ) - 2) + sqr(m_a) * sqr(y(numberOfVariables()-1) - 2);
106
107 for (unsigned i = 1; i < numberOfVariables() - 1; i++) {
108 result += m_a * sqr(y( i ) - 2);
109 }
110
111 value[1] = result / ( sqr(m_a) * numberOfVariables() );
112
113 return value;
114 }
115
116 SearchPointType proposeStartingPoint() const {
117 RealVector x(m_numberOfVariables);
118
119 for (std::size_t i = 0; i < x.size(); i++) {
120 x(i) = random::uni(*mep_rng, -10.0, 10.0);
121 }
122 return x;
123 }
124private:
125 double m_a;
126 RealMatrix m_rotationMatrix;
127 std::size_t m_numberOfVariables;
128};
129}}
