32 // Implementation of the CMA-ES
33 #include <shark/Algorithms/DirectSearch/CMA.h>
34 // Access to benchmark functions
35 #include <shark/ObjectiveFunctions/Benchmarks/Benchmarks.h>
36
37using namespace shark;
38using namespace std;
39
40int main( int argc, char ** argv ) {
41
42 // Adjust the floating-point format to scientific and increase output precision.
43 cout.setf( ios_base::scientific );
44 cout.precision( 10 );
45
46 // Instantiate the problem.
47 benchmarks::Sphere sphere( 2 );
48 // Initialize the optimizer for the objective function instance.
49 CMA cma;
50 cma.setInitialSigma(0.1);// Explicitely set initial global step size.
51 sphere.init();
52 cma.init( sphere, sphere.proposeStartingPoint());
53
54 // Iterate the optimizer until a solution of sufficient quality is found.
55 do {
56 cma.step( sphere );
57
58 // Report information on the optimizer state and the current solution to the console.
59 cout << sphere.evaluationCounter() << " " << cma.solution().value << " " << cma.solution().point << " " << cma.sigma() << endl;
60 } while(cma.solution().value > 1E-20 );
61}
